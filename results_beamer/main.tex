\begin{frame}{Key Equations}
  \begin{align*}
    Q^2 &= -q^2, \quad q = k - k' \\
    x_{Bj} &= \frac{Q^2}{2 p \cdot q} \\
    y &= \frac{p \cdot q}{p \cdot k} \\
    W^2 &= (p + q)^2 \\
    |t| &= |(p - p')^2|
  \end{align*}
\end{frame}
